AC自动机

所谓 AC自动机 ，就是提交上去自动 AC 的代码，是一种有限状态自动机（废话），这篇文章不讲自动机只讲 AC，是进行多模式匹配的利器

KMP很快但只能一对一查找，trie 虽然能多模式匹配，但多次匹配会花费额外时间回溯，模式串很长又会很浪费时间，那么两者能否合成一下？答案是肯定的，AC自动机就是这种类似在 trie 上跑 KMP 的算法（严格来说并不是，但这很方便理解），他利用 fail 失配指针防止进行多余的匹配回溯来优化时间复杂度。

如果我们一个文本串 T和 n 个 模式串 S 如下

T: asherishyouhis
S: he
   her
   his
   is
   you

失配指针

为什么说 \(AC = trie + KMP\) 不太准确呢，trie 部分几乎是一样的，重点在于 KMP 部分，我们来对比一下 AC自动机 的fail和 KMP 的 nex：

共同：都是在匹配失败时用来跳转的指针 不同：nex要求最长公共前缀，fail需要求（当前节点所代表的串）的后缀与（其他节点所代表的串）的前缀的最长，因为 AC 需要对多个模式串进行匹配，fail 可能从一个模式串指向另一个模式串，两者的前缀不同。正因此，AC自动机 在文本串上逐位匹配时，同一位字母可能对应多个模式串。

基础思想

参考 KMP 的 nex 构造思路。 利用之前已经求出的 fail 推导当前节点的 fail 指针。具体使用 BFS 进行层次遍历，设当前节点为 u， 它的父节点为 fa，它们之间连了一条字符 c 的边，且因为是层次遍历，比 u 深度小的节点肯定都已经求过 fail，对于 fail[fa]，有两种情况： 1. 节点 fail[fa] 通过字符 c 连向 w，此时直接令 \(fail[u] = trie[fail[fa]][c]\) 即可，即相当于在 fa 和fail[fa]后面添加了 c 就构成了节点 u 和节点 fail[u],参考图：

  1. 节点 fail[fa] 下面并没有字符 c，那就往上找 fail[fail[fa]],重复上述操作直到 fail 指向根节点，此时就只能让 \(fail[u] = root\) 但是本文章没有考虑这种情况而是利用空节点进行跳转，具体内容一会再说。

上述操作即为 fail 指针的建立。

图解fail：

第一层肯定是指向根节点，不解释

3 的父节点 2 的 fail 并没有一个字符 c 后继，继续往上倒腾，到了根节点，直接令 \(fail[3] = 1\) 即连到根节点。 5 的父节点 2 的 fail 恰好有一个字符 i 的后继，令 \(fail[5] = trie[fail[2]][i - 'a']\) 即连到 7。

接下来的连接同上，最终会长成这个样：

Code： 

void getFail() {
    queue<int> q;//队列辅助宽搜
    fail[0] = 0;//初始化根节点
    for(int i = 0; i < 26; ++i) {
        if(trie[0][i]) {//如果有这个字母
            fail[trie[0][i]] = 0;//fail就是根节点
            q.push(trie[0][i]);
        }
    }//遍历第一层单词开头
    while(!q.empty()) {
        int now = q.front();
        q.pop();
        for(int i = 0; i < 26; ++i) {
            if(trie[now][i]) {
                fail[trie[now][i]] = trie[fail[now]][i];
            //即使trie[fail[now]][i]是个空节点，但是空节点也连了一条fail，所以不用往上找
                q.push(trie[now][i]);//存在一个节点，就连fail，等待扩展
            } else {
                trie[now][i] = trie[fail[now]][i];
            //如果当前now下没有字符i（空节点）则往上找i
            }
        }
    }
}

P3808 【模板】AC 自动机（简单版） 

#include <bits/stdc++.h>
using namespace std;

int n;
string s, str;

int tot = 0;
int fail[1000010], trie[1000010][27], cnt[1000010];

void add(string s) {
    int p = 0;
    for(int i = 0, len = s.size(); i < len; ++i) {
        int num = s[i] - 'a';
        if(!trie[p][num]) trie[p][num] = ++tot;
        p = trie[p][num];
    }
    ++cnt[p];
}
void getFail() {
    queue<int> q;
    fail[0] = 0;
    for(int i = 0; i < 26; ++i) {
        if(trie[0][i]) {
            fail[trie[0][i]] = 0;
            q.push(trie[0][i]);
        }
    }
    while(!q.empty()) {
        int now = q.front();
        q.pop();
        for(int i = 0; i < 26; ++i) {
            if(trie[now][i]) {
                fail[trie[now][i]] = trie[fail[now]][i];
                q.push(trie[now][i]);
            } else {
                trie[now][i] = trie[fail[now]][i];
            }
        }
    }
}
void query(string s) {
    int now = 0, ans = 0;
    for(int i = 0, len = s.size(); i < len; ++i) {//在文本串上遍历
        now = trie[now][s[i] - 'a'];
        for(int j = now; j && cnt[j] != -1/*存在这个节点且该节点没有被遍历过*/; j = fail[j]/*暴力跳fail边*/) {
            ans += cnt[j];
            cnt[j] = -1;
        }
    }
    printf("%d", ans);
}
int main() {
    cin >> n;
    for(int i = 1; i <= n; ++i) {
        cin >> s;
        add(s);
    }
    getFail();
    cin >> str;
    query(str);
    return 0;
}

P3796 【模板】AC 自动机（加强版） 变化不大，只是变成了多组数据，以及输出在文本串上出现次数最多的模式串。 

#include <bits/stdc++.h>
using namespace std;

int n;
string s, str;

int tot = 0;
string ask[153];
int fail[1000010], cnt[1000010], trie[1000010][26];

unordered_map<string, int> mp;

void clear() {//清空
    tot = 0;
    mp.clear();
    memset(cnt, 0, sizeof(cnt));
    memset(fail, 0, sizeof(fail));
    memset(trie, 0, sizeof(trie));
}
void add(string s) {
    int p = 0, len = s.size();
    for(int i = 0; i < len; ++i) {
        int num = s[i] - 'a';
        if(!trie[p][num]) trie[p][num] = ++tot;
        p = trie[p][num];
    }
    cnt[p] = len; //根节点到 p 点的长度即单词长度
}
void getFail() {
    queue<int> q;
    fail[0] = 0;
    for(int i = 0; i < 26; ++i) {
        if(trie[0][i]) {
            fail[trie[0][i]] = 0;
            q.push(trie[0][i]);
        }
    }
    while(!q.empty()) {
        int now = q.front();
        q.pop();
        for(int i = 0; i < 26; ++i) {
            if(trie[now][i]) {
                fail[trie[now][i]] = trie[fail[now]][i];
                q.push(trie[now][i]);
            } else {
                trie[now][i] = trie[fail[now]][i];
            }
        }
    }
}
void query(string s) {
    int now = 0, ans = -1;
    for(int i = 0, len = s.size(); i < len; ++i) {
        now = trie[now][s[i] - 'a'];
        for(int j = now; j && cnt[j] != -1; j = fail[j]) {
            if(cnt[j])
                ++mp[s.substr(i - cnt[j] + 1, cnt[j])];//从原串上把这个字串截取下来统计次数
        }
    }
    for(auto it = mp.begin(); it != mp.end(); ++it)
        ans = max(ans, it->second);//迭代遍历找最大值
    printf("%d\n", ans);
    for(int i = 1;i <= n; ++i)
        if(mp[ask[i]] == ans)//看谁出现的次数最多
            cout << ask[i] <<endl;
}
int main() {
    // freopen("in.txt", "r", stdin);
    // freopen("out.txt","w",stdout);
    while(1) {
        cin >> n;
        if(n == 0) return 0;
        clear();
        for(int i = 1; i <= n; ++i) {
            cin >> ask[i];
            add(ask[i]);
        }
        getFail();
        cin >> str;
        query(str);
    }
    return 0;
}
P5357 【模板】AC 自动机（二次加强版） 本题需要亿点点小优化（如果您卡常过了才是真的巨）

还是这个图（这个图不太好但是我懒得扣了凑活看一下），6节点连向8节点，8节点连向1节点，如果像模板1和2一样暴力跳fail边，那么对6进行操作时就会接着对 8 和 1 进行操作，同理，在走到 8 节点时就会对 1 进行操作（其实就是暴力跳 fail 的过程），这样 1 就被操作了两次，如果这个树有很多像 aaaaaa...aaa 这样的串，就会无比浪费时间。

现在引入一个标记，当操作 6 节点时打上 1 个标记，操作 8 节点时打上 1 标记，操作 1 节点时再打上一个标记，最后通过 fail 边统计答案，在 8 节点上加上 6 节点的标记数，操作 1 节点时加上 8 节点（此时它已经统计了 6 节点的标记）上的标记，这样这棵树的 fail 只会被遍历一次。

理论成立，现在考虑如何实现，统计答案的顺序是什么呢？ 肯定需要从深层次向浅层次跳，有点像什么？ 拓扑排序！！

用 fail 指针统计入度直接跑拓扑排序统计答案就完事了！

具体修改部分都在代码里： 

#include <bits/stdc++.h>
#define SIZE 200010
using namespace std;

#include <ext/pb_ds/assoc_container.hpp> //只要使用pb_ds库，就要包含
#include <ext/pb_ds/hash_policy.hpp>     //哈希表使用
__gnu_pbds::cc_hash_table<int, int> mp;  // pbds自带的hash表，试图优化一下，但没什么卵用

int n;

string s[SIZE], str;
int tot = 0, flag[SIZE], fail[SIZE], trie[SIZE][26];

int in[SIZE], vis[SIZE], ans[SIZE];

inline void add(string s, int id) {
    int p = 0, len = s.size();
    for(int i = 0; i < len; ++i) {
        int num = s[i] - 'a';
        if(!trie[p][num]) trie[p][num] = ++tot;
        p = trie[p][num];
    }
    if(!flag[p]) flag[p] = id; // 记录p点是某个单词的结尾
    mp[id] = flag[p];          // 第id和单词的结尾编号是p,相当于单词去重，如果 falg[p] 为真说明与之前的串重复
}
inline void getFail() {
    queue<int> q;
    fail[0] = 0;
    for(int i = 0; i < 26; ++i) {
        if(trie[0][i]) {
            fail[trie[0][i]] = 0;
            q.push(trie[0][i]);
        }
    }
    while(!q.empty()) {
        int now = q.front();
        q.pop();
        for(int i = 0; i < 26; ++i) {
            int to = trie[now][i];
            if(to) {
                fail[to] = trie[fail[now]][i];
                ++in[fail[to]]; // 统计入度
                q.push(to);
            } else {
                trie[now][i] = trie[fail[now]][i];
            }
        }
    }
}
inline void query(string str) {
    int now = 0;
    for(int i = 0, len = str.size(); i < len; ++i) {
        now = trie[now][str[i] - 'a'];
        ++ans[now]; // 不跳边直接统计访问次数
    }
}
inline void topsort() {
    queue<int> q;
    for(int i = 1; i <= tot; ++i)
        if(!in[i]) q.push(i); // 没有入度，肯定是最底层的点
    while(!q.empty()) {
        int now = q.front();
        q.pop();
        vis[flag[now]] = ans[now];//统计总访问次数

        int to = fail[now];
        --in[to];
        ans[to] += ans[now]; // 从底层往上加
        if(!in[to]) q.push(to);
    }
}
int main() {
    // freopen("in.txt","r",stdin);
    // freopen("out.txt","w",stdout);
    cin >> n;
    for(int i = 1; i <= n; ++i) {
        cin >> s[i];
        add(s[i], i);
    }
    getFail();
    cin >> str;
    query(str);
    topsort();
    for(int i = 1; i <= n; ++i)
        printf("%d\n", vis[mp[i]]);
    return 0;
}
